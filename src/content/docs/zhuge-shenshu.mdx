    '開': 12, '發': 12, '系': 8, '統': 12, '程': 12, '式': 5,
    '大': 3, '中': 4, '小': 3, '多': 6, '少': 4, '好': 6,
    '壞': 19, '美': 9, '醜': 17, '高': 10, '低': 7, '新': 13,

    '死': 6, '老': 6, '病': 10, '通': 11, '塞': 13, '全': 6,
    '安': 6, '平': 5, '穩': 19, '創': 12, '設': 11, '計': 9,
    '部': 11, '署': 14, '上': 3, '線': 15, '下': 3, '修': 10,
    '復': 12, '優': 17, '化': 4, '道': 13, '德': 15, '經': 13,
    '乾': 11, '坤': 8, '天': 4, '地': 6, '水': 4, '火': 4,
    '雷': 13, '風': 9, '山': 3, '澤': 15, '學': 16, '習': 11
  };

  // 獲取單個字的筆劃 (若不在 map 中則以 code 雜湊演算法估算 4-17 的合理筆劃)
  function getStrokeCount(char) {
    if (!char) return 0;
    if (strokeMap[char]) return strokeMap[char];
    return (char.charCodeAt(0) % 14) + 4;
  }

  // 生成智能程式禪 tip
  function generateDevZen(interp1, interp2) {
    const text = interp1 + interp2;
    if (text.includes("大吉") || text.includes("順遂") || text.includes("遂心")) {
      return "【主線合併大吉】系統架構運行流暢，今日適合進行重要功能上線或程式碼合併（Main branch merge）。保持敏捷進度，勇敢推進！";
    }
    if (text.includes("不吉") || text.includes("不利") || text.includes("非常之災") || text.includes("阻") || text.includes("凶")) {
      return "【限流降級防禦】當前業務邏輯或系統可能存在隱性漏洞或執行緒競爭。今日宜暫緩大規模上線，專注於架構防禦、數據庫備份與安全檢修。";
    }
    if (text.includes("病") || text.includes("醫") || text.includes("藥")) {
      return "【名醫診脈除錯】代碼中可能隱藏著不易察覺的記憶體洩漏或死鎖。此時不應繼續堆疊新功能，應像名醫把脈般，細細排查日誌與 Profiling。";
    }
    if (text.includes("財") || text.includes("利")) {
      return "【資源節能優化】今天極其適合優化雲端伺服器（Cloud Server）開銷，重構高耗能查詢，增加 Redis 快取。優化算力就是開源節流！";
    }
    if (text.includes("爭") || text.includes("訟") || text.includes("口舌")) {
      return "【共識解耦溝通】今日最需注意開發團隊內部的接口規範溝通。避免未經協調擅自修改共用底層 API，多開 10 分鐘對齊會議可避百禍。";
    }
    if (text.includes("守") || text.includes("等待") || text.includes("緩") || text.includes("耐")) {
      return "【代碼重構沉澱】目前並非大張旗鼓推進新功能的好時機。今日適合靜下心來撰寫單元測試、補齊文檔，消除技術債，沉澱核心架構。";
    }
    // 預設
    return "【守常行健】保持模組高內聚、低耦合的設計規範。不越權，不進行過度設計（Over-engineering），守住系統安全運行的底線。";
  }

  window.addEventListener('DOMContentLoaded', () => {
    // 獲取諸葛數據
    const holder = document.getElementById('zhuge-data-holder');
    if (!holder) return;
    const zhugeData = JSON.parse(holder.getAttribute('data-zhuge'));

    // DOM 元素
    const tabWord = document.getElementById('tabWord');
    const tabDraw = document.getElementById('tabDraw');
    const sectionWord = document.getElementById('sectionWord');
    const sectionDraw = document.getElementById('sectionDraw');

    const zhugeInput = document.getElementById('zhugeInput');
    const calcBtn = document.getElementById('calcBtn');
    const drawBtn = document.getElementById('drawBtn');
    const inputError = document.getElementById('inputError');

    const scrollScene = document.getElementById('scrollScene');
    const scrollPaper = document.getElementById('scrollPaper');
    const lotTag = document.getElementById('lotTag');
    const lotStatus = document.getElementById('lotStatus');
    const lotPoem = document.getElementById('lotPoem');
    const lotInterp1 = document.getElementById('lotInterp1');
    const lotInterp2 = document.getElementById('lotInterp2');
    const lotDevZen = document.getElementById('lotDevZen');

    // Tab 切換
    tabWord.addEventListener('click', () => {
      tabWord.classList.add('active');
      tabDraw.classList.remove('active');
      sectionWord.classList.add('active-section');
      sectionDraw.classList.remove('active-section');
      scrollScene.style.display = 'none';
      scrollPaper.classList.remove('unfolded');
    });

    tabDraw.addEventListener('click', () => {
      tabDraw.classList.add('active');
      tabWord.classList.remove('active');
      sectionDraw.classList.add('active-section');
      sectionWord.classList.remove('active-section');
      scrollScene.style.display = 'none';
      scrollPaper.classList.remove('unfolded');
    });

    // 渲染錦囊卷軸
    function renderScroll(lot) {
      scrollScene.style.display = 'block';
      scrollPaper.classList.remove('unfolded');

      // 填充內容
      lotTag.textContent = '諸葛神數・第 ' + lot.id + ' 籤';
      
      // 提取吉凶標題
      let status = '平平';
      if (lot.interp1.includes('大吉')) status = '🌟 大吉大利';
      else if (lot.interp1.includes('吉')) status = '✨ 吉祥如意';
      else if (lot.interp1.includes('不利') || lot.interp1.includes('不吉') || lot.interp1.includes('不祥')) status = '⚠️ 警惕守防';
      else if (lot.interp1.includes('平平')) status = '⚖️ 穩健平順';

      lotStatus.textContent = status;
      lotPoem.textContent = lot.poem ? '「' + lot.poem + '」' : '「' + lot.interp1.split('，')[0] + '」';
      lotInterp1.textContent = lot.interp1;
      lotInterp2.textContent = lot.interp2 || '古典註解待參透，以平靜心態處之即可。';
      lotDevZen.textContent = generateDevZen(lot.interp1, lot.interp2);

      // 動畫展開
      setTimeout(() => {
        scrollPaper.classList.add('unfolded');
      }, 150);
    }

    // 測字占卜
    calcBtn.addEventListener('click', () => {
      const text = zhugeInput.value.trim();
      inputError.textContent = '';

      if (text.length !== 3) {
        inputError.textContent = '請務必輸入恰好三個漢字！';
        return;
      }

      // 檢查是否都為漢字
      const isChinese = /^[\\u4e00-\\u9fa5]{3}$/.test(text);
      if (!isChinese) {
        // 放寬一點點以相容有些系統特殊字元，但主要字還是要計算
        if (text.length !== 3) {
          inputError.textContent = '請輸入三個中文漢字！';
          return;
        }
      }

      // 計算筆劃
      const A = getStrokeCount(text[0]);
      const B = getStrokeCount(text[1]);
      const C = getStrokeCount(text[2]);

      // 傳統轉換算法：
      // 百位為 A % 10 (10為0)
      // 十位為 B % 10 (10為0)
      // 個位為 C % 10 (10為0)
      const h = A % 10;
      const t = B % 10;
      const u = C % 10;

      let X = h * 100 + t * 10 + u;
      
      //  modulo 384
      if (X > 384) {
        X = X % 384;
      }
      if (X === 0) {
        X = 384;
      }

      // 查找對應籤詩
      const lot = zhugeData.find(item => item.id === X);
      if (lot) {
        renderScroll(lot);
      } else {
        // 防呆，隨機選一個
        const randX = Math.floor(Math.random() * 384) + 1;
        renderScroll(zhugeData[randX - 1]);
      }
    });

    // 隨機抽取今日錦囊
    drawBtn.addEventListener('click', () => {
      const randIndex = Math.floor(Math.random() * zhugeData.length);
      const lot = zhugeData[randIndex];
      renderScroll(lot);
    });
  });
